\subsection{Синтез \(H_\infty\)-регулятора по выходу.}
\begin{equation}
    \begin{cases}
        AP + PA^T + B_1B_1^T - PC_1^T(D_1D_1^T)^{-1}C_1P = 0 \\
        L = - PC_1^T(D_1^TD_1)^{-1} \\
        A^TQ + QA + C_2^TC_2 - QB_2(D_2^TD_2)^{-1}B_2^TQ = 0 \\
        K = -(D_2^TD_2)^{-1}B_2^TQ
    \end{cases}
\end{equation}

Представим систему в виде:
\begin{equation}
    \begin{cases}
        \begin{bmatrix}
            \dot{x} \\ \dot{e}
        \end{bmatrix} =
        \begin{bmatrix}
            A + B_2K & -B_2K \\
            -(LD_1B_1)\gamma^{-2}B_1^TQ & A + LC_1 + (LD_1B_1)\gamma^{-2}B_1^TQ
        \end{bmatrix} \begin{bmatrix}
            x \\ e
        \end{bmatrix} +
        \begin{bmatrix}
            B_1 \\
            LD_1 + B_1
        \end{bmatrix}w \\
        z = \begin{bmatrix}
            C_2 +D_2K & -D_2K
        \end{bmatrix}\begin{bmatrix}
            x \\ e
        \end{bmatrix}
    \end{cases}
\end{equation}
